def knapsack(items, capacity):
    """Solves the knapsack problem using a greedy algorithm.

    Args:
        items: A list of items, where each item is a tuple of (weight, 
            value).
        capacity: The knapsack capacity.

    Returns:
        The maximum value that can be put in the knapsack.
    """

    # Sort the items in decreasing order of value per weight.
    items.sort(key=lambda item: item[1] / item[0], reverse=True)

    # Initialize the knapsack value.
    knapsack_value = 0

    # Iterate over the items and add them to the knapsack until we reach 
    the capacity.
    for item in items:
        if knapsack_value + item[1] <= capacity:
            knapsack_value += item[1]
            capacity -= item[0]
        else:
            break
    return knapsack_value
